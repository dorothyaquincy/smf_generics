SELECT 
QMGR as Queue_Namager, 
Base_Name as Base_Name,
MAX(CF_Structure)  AS CF_Structure,
CHAR(SUM(Open_Count)) AS Open_Count,
CHAR(SUM(OPEN_ET_US)) AS Open_Total_ET,
CHAR(SUM(OPEN_CT_US)) AS Open_Total_CT,
CHAR(SUM(Close_Count)) AS Close_Count,
CHAR(SUM(Close_ET_US)) AS Close_Total_ET,
CHAR(SUM(Close_CT_US)) AS Close_Total_CT,
CHAR(SUM(Get_Count)) As Get_Count,
CHAR(SUM(GET_ET_US)) AS GET_Total_ET,
CHAR(SUM(GET_CT_US)) AS GET_Total_CT,
CHAR(SUM(Total_Valid_Gets)) AS Total_Valid_Gets,
CHAR(SUM(Total_Bytes_Get)) AS Total_Bytes_Get,
CHAR(SUM(Get_Persistent_Count)) AS Get_Persistent_Count,
CHAR(SUM(Get_Messages_Skipped_Count)) AS Skipped_Count,
CHAR(SUM(Get_Messages_Expired_Count)) AS Expired_Count,
CHAR(SUM(Put_Count)) AS Put_Count,
CHAR(SUM(PUT_ET_US)) AS PUT_Total_ET,
CHAR(SUM(PUT_CT_US)) AS PUT_Total_CT,
CHAR(SUM(Put1_Count)) AS Put1_Count,
CHAR(SUM(PUT1_ET_US)) AS PUT1_Total_ET,
CHAR(SUM(PUT1_CT_US)) AS PUT1_Total_CT,
CHAR(SUM(Total_Valid_Puts)) AS Total_Valid_Puts,
CHAR(SUM(Total_Bytes_Put)) AS Total_Bytes_Put,
CHAR(SUM(PUT_WAITING_GETTER_COUNT)) AS Total_Put2WGET,
CHAR(SUM(PUT1_WAITING_GETTER_COUNT)) AS Total_Put12WGET,
CHAR(SUM(Generated_Msgs)) AS Generated_Msgs,
CHAR(SUM(Put_Persistent_Count)) AS Put_Persistent_Count,
CHAR(SUM(Put1_Persistent_Count)) AS Put1_Persistent_Count,
CHAR(MAX(Max_Depth)) AS Maximum_Depth,
CHAR(MAX(MAX_TIME_ON_QUEUE_US)) AS Maximum_Time_onQ,
CHAR(MIN(MIN_TIME_ON_QUEUE_US)) AS Minimum_Time_onQ,
CHAR(SUM(Inq_Count)) AS Inq_Count,
CHAR(SUM(Set_Count)) AS Set_Count,
 '2' AS Row_ID 
FROM MQSMF.WQ
Where CF_Structure <> ' ' 
Group by Base_Name, QMGR
UNION
SELECT
' Queue Manager ',
' Base Queue Name',
' Coupling Facility Structure',
' Total Opens',
' Total Open Elapsed Time',
' Total Open CPU Time',
' Total Closes',
' Total Close Elapsed Time',
' Total Close CPU Time',
' Total MQGET Requests',
' Total MQGET Elapsed Time',
' Total MQGET CPU Time',
' Total Valid Gets',
' Total Get Bytes',
' Total Persistent MQGETs',
' Total Messages Skipped',
' Total Messages Expired',
' Total MQPUT Requests',
' Total MQPUT Elapsed Time',
' Total MQPUT CPU Time',
' Total MQPUT1 Requests',
' Total MQPUT1 Elapsed Time',
' Total MQPUT1 CPU Time',
' Total Valid MQPUTs',
' Total Bytes Put ',
' Total Puts to Waiting Getter ',
' Total Put1s to Waiting Getter ',
' Total Generated Messages ',
' Total Persistent MQPUTs',
' Total Persistent MQPUT1s',
' Max Depth on Queue',
' Max Time on Queue ', 
' Min Time on Queue ', 
' Total Inquiries ',
' Total Sets ',
 '1' AS Row_ID 
FROM SYSIBM.SYSDUMMY1 ORDER BY Row_ID, Base_Name
;
